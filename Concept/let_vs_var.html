function LetvsVar() { //letYoung is *not* visible out here for( let letYoung = 0; letYoung < 5; letYoung++ ) { //letYoung is only visible in here (and in the for() parentheses) document.write('in for - letYoung : ' + letYoung); }; for( var varYoung = 0; varYoung < 5; varYoung++ ) { //varYoung visible in here document.write('in for - varYoung : '+ varYoung); }; //letYoung is *not* visible out here //varYoung visible in here try{ document.write('out for - letYoung : ' + letYoung); } catch(e){ document.write(' error : '+ e.message); } try{ document.write('out for - varYoung :' + varYoung); } catch(e){ document.write(' error : ' + e.message); } };
